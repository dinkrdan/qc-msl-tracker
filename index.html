    <label><input type="checkbox" id="shuffleToggle"> Shuffle courts</label>

    <label><input type="checkbox" id="avoidMMFF"> Avoid MM vs FF</label>

  <div id="benchDisplay"></div>


  <script>

    const players = [

      "Alex (M, 4.03)", "Betty (F, 4.16)", "Charles (M, 3.55)", "Dan (M, 3.67)",

      "Eric (M, 3.86)", "Fiona (F, 4.03)", "Georgina (F, 3.97)", "Henry (M, 3.54)",

      "Ian (M, 3.83)", "James (M, 3.55)", "Kevin (M, 4.03)", "Liam (M, 4.10)",

      "Michael (M, 3.91)", "Nina (F, 4.02)", "Oscar (M, 4.20)", "Peter (M, 4.13)",

      "Quinn (M, 3.58)", "Rita (F, 3.71)", "Steve (M, 3.98)"

    ];


    let courts = {};

    let nextUp = [];

    let bench = [];

    let sessionActive = false;


    function shuffle(array) {

      for (let i = array.length - 1; i > 0; i--) {

        const j = Math.floor(Math.random() * (i + 1));

        [array[i], array[j]] = [array[j], array[i]];

      }

    }


    function parsePlayer(name) {

      const match = name.match(/^(.+?) \((.)\, ([\d.]+)\)$/);

      return match ? {

        name: match[1],

        gender: match[2],

        rating: match[3]

      } : { name, gender: '', rating: '' };

    }


    function formatPlayer(name) {

      const { name: n, gender: g, rating: r } = parsePlayer(name);

      return `<span class="player-name">${n}</span> <span class="player-meta">(${g}, ${r})</span>`;

    }


    function toggleSession() {

      if (!sessionActive) {

        startSession();

        sessionActive = true;

        document.getElementById("sessionButton").textContent = "End Session";

      } else {

        sessionActive = false;

        courts = {};

        nextUp = [];

        bench = [];

        document.getElementById("courtsDisplay").innerHTML = "";

        document.getElementById("nextUpDisplay").innerHTML = "";

        document.getElementById("benchDisplay").innerHTML = "";

        document.getElementById("sessionButton").textContent = "Start Session";

      }

    }


    function startSession() {

      const allPlayers = [...players];

      shuffle(allPlayers);

      courts = {

        "Court 7": allPlayers.splice(0, 4),

        "Court 6": allPlayers.splice(0, 4),

        "Court 5": allPlayers.splice(0, 4)

      };

      nextUp = allPlayers.splice(0, 4);

      bench = allPlayers;

      render();

    }


    function render() {

      const display = document.getElementById("courtsDisplay");

      display.innerHTML = "";

      for (const [court, players] of Object.entries(courts)) {

        if (players.length === 4) {

          const [p1, p2, p3, p4] = players;

          display.innerHTML += `

            <div class="court">

              <h2>${court}</h2>

              <div class="matchup-line">

                ${formatPlayer(p1)} + ${formatPlayer(p2)} vs ${formatPlayer(p3)} + ${formatPlayer(p4)}

              </div>

              <div class="controls">

                <button onclick="endGame('${court}')">End Game</button>

              </div>

            </div>`;

        }

      }


      document.getElementById("nextUpDisplay").innerHTML = `

        <h3>Next Up</h3>

        ${nextUp.map(p => formatPlayer(p)).join(', ')}`;


      document.getElementById("benchDisplay").innerHTML = `

        <h3>Bench</h3>

        ${bench.map(p => formatPlayer(p)).join(', ')}`;

    }


    function endGame(court) {

      const shuffleCourts = document.getElementById("shuffleToggle").checked;

      const endedPlayers = courts[court];

      if (!endedPlayers || endedPlayers.length !== 4) return;


      if (shuffleCourts) {

        courts[court] = courts["Court 6"];

        courts["Court 6"] = courts["Court 5"];

        courts["Court 5"] = nextUp;

      } else {

        courts[court] = nextUp;

      }


      const refill = bench.splice(0, 3);

      const randomOne = endedPlayers[Math.floor(Math.random() * 4)];

      nextUp = [...refill, randomOne];

      bench = [...bench, ...endedPlayers.filter(p => p !== randomOne)];

      render();

    }
